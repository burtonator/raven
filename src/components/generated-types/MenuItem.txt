
export type MenuItemRef = Readonly<{
  MenuItem: MenuItemRefProperties
}>

export type MenuItemRefProperties = Readonly<{
  // The content of the component.
  children: React.ReactNode

  // Override or extend the styles applied to the component.
  classes: object

  // The component used for the root node.
  component: React.ElementType

  // The container component.
  containerComponent: React.ElementType

  // The props used for the container component.
  containerProps: object

  // If true, the menu item will be disabled.
  disabled: boolean

  // If true, the menu item will have keyboard focus.
  focusVisibleClassName: string

  // The URL to link to when the menu item is clicked.
  href: string

  // The component used for the ListItem node.
  ListItemClasses: object

  // The main container element.
  role: string

  // The selected state of the menu item.
  selected: boolean

  // The system to be used.
  tabIndex: number
}>
