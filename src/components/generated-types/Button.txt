
/**
 * Buttons allow users to take actions, and make choices, with a single tap.
 */
export type ElementRef = ButtonRef

export type ButtonRef = Readonly<{
  Button: ButtonRefProperties
}>

export type ButtonRefProperties = Readonly<{
  /**
   * The content of the button.
   */
  children: string

  /**
   * The color of the button. It supports those theme colors that make sense for this component.
   */
  color: 'default' | 'inherit' | 'primary' | 'secondary'

  /**
   * If `true`, the button will be disabled.
   */
  disabled: boolean

  /**
   * If `true`, the button will take up the full width of its container.
   */
  fullWidth: boolean

  /**
   * The size of the button. `small` is equivalent to the dense button styling.
   */
  size: 'small' | 'medium' | 'large'

  /**
   * The variant to use.
   */
  variant: 'text' | 'outlined' | 'contained'
}>
