
/**
 * The Card component is a flexible and extensible content container. It includes options for headers and footers, a wide variety of content, contextual background colors, and powerful display options.
 */
export type ElementRef = CardRef

export type CardRef = Readonly<{
  Card: CardRefProperties
}>

export type CardRefProperties = Readonly<{
  /**
   * If `true`, the card will use raised styling.
   */
  raised: boolean
  /**
   * The elevation of the card. Shadow cast corresponds to the `z-index` of the elevation.
   */
  elevation: number
}>
