
/**
 * Icon buttons are buttons that contain an icon instead of text. They can be used to represent a specific action.
 */
export type IconButtonRef = Readonly<{
  IconButton: IconButtonRefProperties
}>

export type IconButtonRefProperties = Readonly<{
  color?: 'inherit' | 'primary' | 'secondary' | 'default'
  disabled?: boolean
  disableFocusRipple?: boolean
  disableRipple?: boolean
  edge?: 'start' | 'end' | false
  size?: 'small' | 'medium'
}>
