
/**
 * The IconButton component is used to display a button with an icon.
 * It can be used to trigger an action or to navigate between views.
 */
export type ElementRef = IconButtonRef

export type IconButtonRef = Readonly<{
  IconButton: IconButtonRefProperties
}>

export type IconButtonRefProperties = Readonly<{
  /**
   * The color of the button. It supports two values: 'default', 'inherit', 'primary', and 'secondary'.
   */
  color: 'default' | 'inherit' | 'primary' | 'secondary'

  /**
   * If `true`, the button will be disabled.
   */
  disabled: boolean

  /**
   * The edge of the button. It supports two values: 'start', 'end', and 'false'.
   */
  edge: 'start' | 'end' | false

  /**
   * The size of the button. It supports two values: 'small', 'medium', and 'large'.
   */
  size: 'small' | 'medium' | 'large'
}>
