
// TextField component allows users to input text and provides a label and helper text for additional information.
export type ElementRef = TextFieldRef

export type TextFieldRef = Readonly<{
  TextField: TextFieldRefProperties
}>

export type TextFieldRefProperties = Readonly<{
  // The type of input to display such as "text", "email", "password", etc.
  type: string

  // The label text displayed above the input field.
  label: string

  // The helper text displayed below the input field.
  helperText: string

  // If true, the input field will be disabled.
  disabled: boolean

  // If true, the input field will be required.
  required: boolean

  // If true, the input field will be read-only.
  readOnly: boolean

  // If true, the input field will have an error state.
  error: boolean

  // The value of the input field.
  value: string | number

  // The default value of the input field.
  defaultValue: string | number

  // The variant to use for the input field. Can be "standard", "outlined", or "filled".
  variant: 'standard' | 'outlined' | 'filled'

  // The size of the input field. Can be "small" or "medium".
  size: 'small' | 'medium'

  // The width of the input field.
  fullWidth: boolean

  // The margin applied to the input field. Can be "none", "dense", or "normal".
  margin: 'none' | 'dense' | 'normal'

  // The id of the input element.
  id: string

  // The name attribute of the input element.
  name: string

  // The autoComplete attribute of the input element.
  autoComplete: string

  // The autoFocus attribute of the input element.
  autoFocus: boolean

  // The inputMode attribute of the input element.
  inputMode: string

  // The min attribute of the input element.
  min: string

  // The max attribute of the input element.
  max: string

  // The minLength attribute of the input element.
  minLength: number

  // The maxLength attribute of the input element.
  maxLength: number

  // The step attribute of the input element.
  step: string | number

  // The pattern attribute of the input element.
  pattern: string

  // The placeholder attribute of the input element.
  placeholder: string

  // The rows attribute of the input element.
  rows: number

  // The rowsMax attribute of the input element.
  rowsMax: number

  // The rowsMin attribute of the input element.
  rowsMin: number

  // The select attribute of the input element.
  select: boolean

  // The tabIndex attribute of the input element.
  tabIndex: number

  // The inputProps attribute of the input element.
  inputProps: object
}>
