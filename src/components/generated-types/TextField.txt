
export type ElementRef = TextFieldRef

/**
 * TextField is a Material-UI component that provides a user input field with a label and optional helper text.
 * It supports standard text input, as well as select input and multiline input.
 */
export type TextFieldRef = Readonly<{
  TextField: TextFieldRefProperties
}>

export type TextFieldRefProperties = Readonly<{
  /**
   * The label content.
   */
  label: string

  /**
   * The short hint displayed in the input before the user enters a value.
   */
  placeholder: string

  /**
   * The type of input to display such as "password", "email", "number", etc.
   */
  type: string

  /**
   * If `true`, the input will be disabled.
   */
  disabled: boolean

  /**
   * If `true`, the label will be displayed in an error state.
   */
  error: boolean

  /**
   * If `true`, the input will take up the full width of its container.
   */
  fullWidth: boolean

  /**
   * If `true`, the label is hidden when the input is focused and filled.
   */
  shrink: boolean

  /**
   * If `true`, the input will be required.
   */
  required: boolean

  /**
   * The variant to use. Use "outlined" for the default Material Design style.
   */
  variant: 'standard' | 'outlined' | 'filled'

  /**
   * The helper text content.
   */
  helperText: string

  /**
   * If `true`, the helper text is displayed in an error state.
   */
  helperTextError: boolean

  /**
   * If `true`, the helper text is displayed below the input.
   */
  helperTextBelow: boolean

  /**
   * If `true`, the input will be multiline.
   */
  multiline: boolean

  /**
   * Number of rows to display when multiline option is set to true.
   */
  rows: number

  /**
   * Maximum number of rows to display when multiline option is set to true.
   */
  rowsMax: number

  /**
   * If `true`, a textarea element will be rendered instead of an input.
   */
  textarea: boolean

  /**
   * If `true`, the input will have an autoComplete attribute.
   */
  autoComplete: boolean

  /**
   * The value of the input element, required for a controlled component.
   */
  value: string | number

  /**
   * The default value of the input element.
   */
  defaultValue: string | number

  /**
   * The name attribute of the input element.
   */
  name: string

  /**
   * The id of the input element.
   */
  id: string
}>
