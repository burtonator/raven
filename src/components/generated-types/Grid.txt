
export type GridRef = Readonly<{
  Grid: GridRefProperties
}>

export type GridRefProperties = Readonly<{
  // The number of columns.
  cols: number

  // Defines the space between the type 'item' component.
  spacing: number

  // The system grid to use.
  grid: 'default' | 'dense' | 'loose'

  // The number of columns for a specific breakpoint.
  xs: number
  sm: number
  md: number
  lg: number
  xl: number

  // The width of the container.
  containerWidth: 'xs' | 'sm' | 'md' | 'lg' | 'xl'

  // If true, the component will have the flex 'container' behavior.
  container: boolean

  // If true, the component will have the flex 'item' behavior.
  item: boolean

  // If true, the component will have the flex 'wrap' behavior.
  wrap: boolean

  // The direction of the grid items.
  direction: 'row' | 'row-reverse' | 'column' | 'column-reverse'

  // The alignment of the grid items.
  alignItems: 'flex-start' | 'center' | 'flex-end' | 'stretch' | 'baseline'

  // The content alignment of the grid items.
  alignContent: 'flex-start' | 'center' | 'flex-end' | 'space-between' | 'space-around' | 'stretch'

  // The justification of the grid items.
  justifyContent: 'flex-start' | 'center' | 'flex-end' | 'space-between' | 'space-around' | 'space-evenly'
}>
