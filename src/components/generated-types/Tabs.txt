
export type TabsRef = Readonly<{
  Tabs: TabsRefProperties
}>

export type TabsRefProperties = Readonly<{
  action: (actions: TabsActions) => void
  centered: boolean
  indicatorColor: 'secondary' | 'primary'
  onChange: (event: React.ChangeEvent<{}>, value: any) => void
  orientation: 'horizontal' | 'vertical'
  scrollButtons: 'auto' | 'desktop' | 'on' | 'off'
  selectionFollowsFocus: boolean
  TabIndicatorProps: Partial<TabIndicatorProps>
  textColor: 'secondary' | 'primary' | 'inherit'
  value: any
  variant: 'standard' | 'scrollable' | 'fullWidth'
}>

export type TabsActions = {
  updateIndicator: () => void
}

export type TabIndicatorProps = {
  color: 'secondary' | 'primary' | string
  style: React.CSSProperties
  orientation: 'horizontal' | 'vertical'
}
