
/**
 * The top App bar provides content and actions related to the current screen.
 * It's used for branding, screen titles, navigation, and actions.
 * It can transform into a contextual action bar or be used as a navbar.
 */
export type ElementRef = AppBarRef

export type AppBarRef = Readonly<{
  AppBar: AppBarRefProperties
}>

export type AppBarRefProperties = Readonly<{
  /**
   * The color of the component. It supports those theme colors that make sense for this component.
   */
  color?: 'default' | 'inherit' | 'primary' | 'secondary'

  /**
   * The positioning type. The behavior of the different options is described in the
   * [position](https://mui.com/components/app-bar/#position) section of the documentation.
   */
  position?: 'fixed' | 'absolute' | 'sticky' | 'static' | 'relative'
}>
