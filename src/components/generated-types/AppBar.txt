
/**
 * The AppBar component is a container for positioning navigation, branding, and actions at the top of the screen.
 */
export type ElementRef = AppBarRef

export type AppBarRef = Readonly<{
  AppBar: AppBarRefProperties
}>

export type AppBarRefProperties = Readonly<{
  /**
   * The color of the AppBar. It supports those theme colors that make sense for this component.
   */
  color: 'default' | 'inherit' | 'primary' | 'secondary' | 'transparent'

  /**
   * The positioning type. The behavior of the different options is described in the MDN web docs.
   * https://developer.mozilla.org/en-US/docs/Learn/CSS/CSS_layout/Positioning
   * Note: `sticky` is not universally supported and will fall back to `static` when unavailable.
   */
  position: 'fixed' | 'absolute' | 'sticky' | 'static' | 'relative'

  /**
   * The variant to use.
   */
  variant: 'regular' | 'dense'
}>
