
/**
 * The Drawer component is a dismissible or permanently visible panel used for navigation in responsive web applications.
 */
export type ElementRef = DrawerRef

export type DrawerRef = Readonly<{
  Drawer: DrawerRefProperties
}>

export type DrawerRefProperties = Readonly<{
  anchor: 'left' | 'top' | 'right' | 'bottom'
  elevation: number
  ModalProps: Partial<ModalRefProperties>
  open: boolean
  PaperProps: Partial<PaperRefProperties>
  SlideProps: Partial<SlideRefProperties>
  transitionDuration: number | { enter?: number; exit?: number }
  variant: 'permanent' | 'persistent' | 'temporary'
}>
