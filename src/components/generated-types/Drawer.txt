
// Drawer component is a responsive drawer that can be permanently visible, or opened and closed by tapping a navigation menu icon or when the user swipes.
export type ElementRef = DrawerRef

export type DrawerRef = Readonly<{
  Drawer: DrawerRefProperties
}>

export type DrawerRefProperties = Readonly<{
  anchor: 'left' | 'top' | 'right' | 'bottom'
  elevation: number
  ModalProps: object
  open: boolean
  PaperProps: object
  SlideProps: object
  variant: 'permanent' | 'persistent' | 'temporary'
}>
