
export type MenuRef = Readonly<{
  Menu: MenuRefProperties
}>

export type MenuRefProperties = Readonly<{
  anchorEl: HTMLElement | null
  anchorOrigin: {
    vertical: 'top' | 'center' | 'bottom'
    horizontal: 'left' | 'center' | 'right'
  }
  anchorPosition: {
    top: number
    left: number
  }
  autoFocus: boolean
  disableAutoFocusItem: boolean
  elevation: number
  getContentAnchorEl: HTMLElement | null
  id: string
  keepMounted: boolean
  MenuListProps: Partial<MenuListRefProperties>
  onClose: (event: React.SyntheticEvent, reason: 'backdropClick' | 'escapeKeyDown') => void
  onEnter: (node: HTMLElement) => void
  onEntered: (node: HTMLElement) => void
  onEntering: (node: HTMLElement) => void
  onExit: (node: HTMLElement) => void
  onExited: (node: HTMLElement) => void
  onExiting: (node: HTMLElement) => void
  open: boolean
  PaperProps: Partial<PaperRefProperties>
  PopoverClasses: Partial<PopoverRefProperties>
  transformOrigin: {
    vertical: 'top' | 'center' | 'bottom'
    horizontal: 'left' | 'center' | 'right'
  }
  TransitionComponent: React.ComponentType
  transitionDuration: number | { enter?: number; exit?: number }
  TransitionProps: Partial<TransitionRefProperties>
}>
