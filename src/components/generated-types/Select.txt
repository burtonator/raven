
/**
 * The Select component is used for collecting user provided information from a list of options.
 */
export type ElementRef = SelectRef

export type SelectRef = Readonly<{
  Select: SelectRefProperties
}>

export type SelectRefProperties = Readonly<{
  /**
   * The value of the `input` element, required for a controlled component.
   */
  value: any

  /**
   * If `true`, the component is disabled.
   */
  disabled?: boolean

  /**
   * If `true`, the input element will be required.
   */
  required?: boolean

  /**
   * The variant to use.
   */
  variant?: 'standard' | 'outlined' | 'filled'

  /**
   * The width of the input.
   */
  inputWidth?: number

  /**
   * The label of the input.
   */
  label?: string

  /**
   * The placeholder of the input.
   */
  placeholder?: string

  /**
   * The `Menu` content.
   */
  MenuProps?: object

  /**
   * The `input` content.
   */
  InputProps?: object

  /**
   * The `InputLabel` content.
   */
  InputLabelProps?: object

  /**
   * The `FormControl` content.
   */
  FormControlProps?: object

  /**
   * The `FormHelperText` content.
   */
  FormHelperTextProps?: object

  /**
   * The `Select` content.
   */
  SelectProps?: object

  /**
   * The `MenuItem` content.
   */
  MenuItemProps?: object
}>
