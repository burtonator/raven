
/**
 * The Toolbar component is a container for grouping and organizing UI elements.
 * It can be used as a standalone component or as part of other components like AppBar.
 */
export type ElementRef = ToolbarRef

export type ToolbarRef = Readonly<{
  Toolbar: ToolbarRefProperties
}>

export type ToolbarRefProperties = Readonly<{
  /**
   * The variant to use. It can be 'regular', 'dense', or 'prominent'.
   */
  variant: 'regular' | 'dense' | 'prominent'

  /**
   * If true, the toolbar will have gutters.
   */
  disableGutters: boolean
}>
